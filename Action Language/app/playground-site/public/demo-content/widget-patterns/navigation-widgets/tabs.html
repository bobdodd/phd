Complexity: HIGH 5 Components WCAG 1.3.1, 2.1.1, 4.1.2

About the Tabs Pattern

The tabs pattern allows users to switch between multiple panels of content. A complete implementation requires:

  1. Tablist container with role="tablist"
  2. Tab buttons with role="tab", aria-selected, and aria-controls
  3. Tab panels with role="tabpanel" and aria-labelledby
  4. Keyboard navigation: Arrow keys navigate tabs, Tab enters panel
  5. State management: aria-selected updates, roving tabindex

// JavaScript - Arrow Key Navigation
const tablist = document.querySelector('[role="tablist"]');
const tabs = Array.from(tablist.querySelectorAll('[role="tab"]'));
const panels = Array.from(document.querySelectorAll('[role="tabpanel"]'));

tablist.addEventListener('keydown', (e) => {
  const current = tabs.indexOf(document.activeElement);
  let next;

  if (e.key === 'ArrowRight') {
    next = (current + 1) % tabs.length;
  } else if (e.key === 'ArrowLeft') {
    next = (current - 1 + tabs.length) % tabs.length;
  } else if (e.key === 'Home') {
    next = 0;
  } else if (e.key === 'End') {
    next = tabs.length - 1;
  } else {
    return;
  }

  e.preventDefault();
  activateTab(tabs[next]);
});

// Activate Tab Function
function activateTab(tab) {
  const index = tabs.indexOf(tab);

  // Update tabs
  tabs.forEach((t, i) => {
    t.setAttribute('aria-selected', i === index);
    t.setAttribute('tabindex', i === index ? '0' : '-1');
  });

  // Update panels
  panels.forEach((p, i) => {
    p.hidden = i !== index;
  });

  tab.focus();
}

// Click handler
tabs.forEach(tab => {
  tab.addEventListener('click', () => activateTab(tab));
});
